{{- /* series-nav.html — rendered inside _default/single.html when the page has a `series` front-matter field. Shows "Part N of M" plus prev/next siblings within the same series, ordered by date. Renders nothing if series is unset or has only one entry. */ -}} {{- with .Params.series -}} {{- $name := . -}} {{- $term := index site.Taxonomies.series ($name | urlize) -}} {{- with $term -}} {{- $pages := .Pages.ByDate -}} {{- if gt (len $pages) 1 -}} {{- $cur := $.Page -}} {{- $idx := -1 -}} {{- range $i, $p := $pages -}} {{- if eq $p.RelPermalink $cur.RelPermalink -}}{{- $idx = $i -}}{{- end -}} {{- end -}} {{- if ge $idx 0 -}} {{- $total := len $pages -}} {{- $prev := "" -}} {{- $next := "" -}} {{- if gt $idx 0 -}}{{- $prev = index $pages (sub $idx 1) -}}{{- end -}} {{- if lt $idx (sub $total 1) -}}{{- $next = index $pages (add $idx 1) -}}{{- end -}}
SERIES {{ $name }} · Part {{ add $idx 1 }} of {{ $total }}
{{- with $prev }} ← prev {{ .Title }} {{- else }} ← {{- end }} {{- with $next }} next → {{ .Title }} {{- else }} → {{- end }}
{{- end -}} {{- end -}} {{- end -}} {{- end -}}
