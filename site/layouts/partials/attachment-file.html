{{/* attachment-file — Phase 6 generic file card. Catch-all for MIME types we don't have a richer preview for. Renders a glyph, filename, byte size, and a Download Original link. The glyph is picked from a small static map keyed off the top-level MIME bucket. */}} {{- $m := .media -}} {{- $caption := .caption -}} {{- $mime := lower (default "" $m.mime_type) -}} {{- $type := default "other" $m.type -}} {{- $glyph := "FILE" -}} {{- if eq $type "image" -}}{{- $glyph = "IMG" -}} {{- else if eq $type "video" -}}{{- $glyph = "VID" -}} {{- else if eq $type "audio" -}}{{- $glyph = "AUD" -}} {{- else if eq $type "document" -}}{{- $glyph = "DOC" -}} {{- else if eq $type "archive" -}}{{- $glyph = "ZIP" -}} {{- else if hasPrefix $mime "text/" -}}{{- $glyph = "TXT" -}} {{- end -}} {{- $size := default 0 $m.size -}}
